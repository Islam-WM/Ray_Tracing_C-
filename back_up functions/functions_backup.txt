bool Ray::compute_closest_object(vector<Object*> objects, double& distance, Object* closest_obj){ // computes closest object and distance
    double distance_to_closest_obj = 1000.0; //maximum length of a ray
    // Object closest_obj;
    bool hit = false;
    for(Object* obj : objects){
        //cout<<"hi"<<endl;
        double radius = obj->get_scale()[0]; // scale is {radius,radius,radius} for sphere, therefore we can just take the first element [0]

        Vector3d Q = location.vect() - obj->get_location();

        double a = rotation.dot(rotation);      // should be = 1
        double b = 2* Q.dot(rotation);
        double c = Q.dot(Q) - radius*radius;
        double d = b*b - 4*a*c;  // discriminant of quadratic

    
        if(d >= 0){
            //this means that there are intersections with the sphere

            double dist = findRoots(a, b, c); // returns the smallest root ie. the length of the ray to the sphere

            if(dist >= 0 && dist <= distance_to_closest_obj){
                hit=true;
                distance_to_closest_obj = dist;
                closest_obj = obj;
                cout<<obj->color[0]<<endl;
                cout<<closest_obj->color[0]<<endl;
                //return_color = return_color/distance;
            }
        }
    }
    distance = distance_to_closest_obj;
    return hit;
}
